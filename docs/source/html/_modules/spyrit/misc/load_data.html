def load_data_recon_3D(Path_files, list_files, Nl, Nc, Nh):
    Data = np.zeros((Nl, Nc, Nh))

    for i in range(0, 2 * Nh, 2):
        Data[:, :, i // 2] = np.rot90(
            np.array(Image.open(Path_files + list_files[i]))
        ) - np.rot90(np.array(Image.open(Path_files + list_files[i + 1])))

    return Data
