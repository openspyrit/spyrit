[docs]def expend_vect(Vect,N1,N2): # Expened a vectors of siez N1 to N2
    V_out = np.zeros(N2)
    S = int(N2/N1)
    j = 0
    ad = 0
    for i in range(N1):
        for j in range(0,S):
            V_out[i+j+ad] = Vect[i]
        ad += S-1
    return(V_out)

    Mres = np.zeros((Nl,Nc))

[docs]def stack_depth_matrice(Mat,Nl,Nc,Nd): # Stack a 3 by 3 matrix along its third dimensions
    M_out = np.zeros((Nl,Nc))

        M_out += Mat[:,:,i]
